343 template <class T>
344 bool CUtlCachedFileData<T>::IsUpToDate()
345 {
346  // Don't call Init/Shutdown if using this method!!!
347  Assert( !m_bInitialized );
348 
349  if ( !m_pszRepositoryFileName || !m_pszRepositoryFileName[ 0 ] )
350  {
351  Error( "CUtlCachedFileData: Can't IsUpToDate, no repository file specified." );
352  return false;
353  }
354 
355  // Always compute meta checksum
356  m_uCurrentMetaChecksum = m_pfnMetaChecksum ? (*m_pfnMetaChecksum)() : 0;
357 
358  FileHandle_t fh;
359 
360  fh = filesystem->Open( m_pszRepositoryFileName, "rb", "MOD" );
361  if ( fh == FILESYSTEM_INVALID_HANDLE )
362  {
363  return false;
364  }
365 
366  // Version data is in first 12 bytes of file
367  byte header[ 12 ];
368  filesystem->Read( header, sizeof( header ), fh );
369  filesystem->Close( fh );
370 
371  int cacheversion = *( int *)&header[ 0 ];
372 
373  if ( UTL_CACHE_SYSTEM_VERSION != cacheversion )
374  {
375  DevMsg( "Discarding repository '%s' due to cache system version change\n", m_pszRepositoryFileName );
376  Assert( !m_bReadOnly );
377  if ( !m_bReadOnly )
378  {
379  filesystem->RemoveFile( m_pszRepositoryFileName, "MOD" );
380  }
381  return false;
382  }
383 
384  // Now parse data from the buffer
385  int version = *( int *)&header[ 4 ];
386  if ( version != m_nVersion )
387  {
388  DevMsg( "Discarding repository '%s' due to version change\n", m_pszRepositoryFileName );
389  Assert( !m_bReadOnly );
390  if ( !m_bReadOnly )
391  {
392  filesystem->RemoveFile( m_pszRepositoryFileName, "MOD" );
393  }
394  return false;
395  }
396 
397  // This is a checksum based on any meta data files which the cache depends on (supplied by a passed in
398  // meta data function
399  unsigned int cache_meta_checksum = (unsigned int)*( int *)&header[ 8 ];
400 
401  if ( cache_meta_checksum != m_uCurrentMetaChecksum )
402  {
403  DevMsg( "Discarding repository '%s' due to meta checksum change\n", m_pszRepositoryFileName );
404  Assert( !m_bReadOnly );
405  if ( !m_bReadOnly )
406  {
407  filesystem->RemoveFile( m_pszRepositoryFileName, "MOD" );
408  }
409  return false;
410  }
411 
412  // Looks valid
413  return true;
414 }

656 template <class T>
657 void CUtlCachedFileData<T>::Save()
658 {
659  char path[ 512 ];
660  Q_strncpy( path, m_pszRepositoryFileName, sizeof( path ) );
661  Q_StripFilename( path );
662 
663  filesystem->CreateDirHierarchy( path, "MOD" );
664 
665  if ( filesystem->FileExists( m_pszRepositoryFileName, "MOD" ) &&
666  !filesystem->IsFileWritable( m_pszRepositoryFileName, "MOD" ) )
667  {
668  filesystem->SetFileWritable( m_pszRepositoryFileName, true, "MOD" );
669  }
670 
671  // Now write to file
672  FileHandle_t fh;
673  fh = filesystem->Open( m_pszRepositoryFileName, "wb" );
674  if ( FILESYSTEM_INVALID_HANDLE == fh )
675  {
676  Warning( "Unable to persist cache '%s', check file permissions\n", m_pszRepositoryFileName );
677  }
678  else
679  {
680  SetDirty( false );
681 
682  int v = UTL_CACHE_SYSTEM_VERSION;
683  filesystem->Write( &v, sizeof( v ), fh );
684  v = m_nVersion;
685  filesystem->Write( &v, sizeof( v ), fh );
686  v = (int)m_uCurrentMetaChecksum;
687  filesystem->Write( &v, sizeof( v ), fh );
688 
689  // Element count
690  int c = Count();
691 
692  filesystem->Write( &c, sizeof( c ), fh );
693 
694  // Save repository back out to disk...
695  CUtlBuffer buf( 0, 0, false );
696 
697  for ( int i = m_Elements.FirstInorder(); i != m_Elements.InvalidIndex(); i = m_Elements.NextInorder( i ) )
698  {
699  buf.SeekPut( CUtlBuffer::SEEK_HEAD, 0 );
700 
701  ElementType_t& element = m_Elements[ i ];
702 
703  char fn[ 512 ];
704  filesystem->String( element.handle, fn, sizeof( fn ) );
705 
706  buf.PutString( fn );
707  buf.PutInt( element.fileinfo );
708 
709  Assert( element.dataIndex != m_Data.InvalidIndex() );
710 
711  T *data = m_Data[ element.dataIndex ];
712 
713  Assert( data );
714 
715  ((IBaseCacheInfo *)data)->Save( buf );
716 
717  int bufsize = buf.TellPut();
718  filesystem->Write( &bufsize, sizeof( bufsize ), fh );
719  filesystem->Write( buf.Base(), bufsize, fh );
720  }
721 
722  filesystem->Close( fh );
723  }
724 
725  if ( m_bSaveManifest )
726  {
727  SaveManifest();
728  }
729 }

935  for ( listEnd = listStart; listEnd < list.Count(); listEnd++ )
936  {

949  bCheck = false;

955  }
956 
957  for ( i = listStart; i < listEnd; i++ )
958  {
959  ElementType_t& element = m_Elements[ list[i].index ];
960

966  }
